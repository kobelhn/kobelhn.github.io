# let/const

  • 不存在变量提升，（同一个块级作用域｜函数作用域内）不允许重复声明
  • 暂时性死区：在代码块内，使用let/const声明变量之前，该变量都是不可用的
  • let/const为JavaScript新增了块级作用域；块级作用域内的函数声明类似于let/const
  • const声明常量，不可修改；本质：const实际上保证的并不是变量的值不得改动，而是变量指向的那个内存地址所保存的数据不得改动，对于基本类型数据，值就保存在变量指向的那个内存地址，因此等同于常量。对于引用类型，变量指向的内存地址保存的是一个指向实际数据的指针，const只能保证这个指针是固定的，至于它指向的数据结构是不是可变的，就完全不能控制了。
const foo = {};
foo.prop = 123;
console.log(foo.prop); // 123
foo = {}; // TypeError: "foo" is read-only
let a = 1;
function test() {
  let a = 2;
  console.log(a);
}
test(); // 2
console.log(a); // 1

# 变量的解构赋值

  • ES6允许按照一定的模式，从数组和对象中提取值，对变量进行赋值，这被称为解构。
  • 本质上，这种解构属于“模式匹配”，只要等号两边的模式相同，左边的变量就会被赋予对应的值。
  • 用途：https://es6.ruanyifeng.com/#docs/destructuring
// 数组
let [a, b, c] = [1, 2, 3];
let [d = 0, [e], f] = [1, [2], 3];
// 对象
let {g, h, i} = { g: 1, h: 2, i: 3 };
let {j: [k, {l = false}]} = {j: [2, {l: true}]};
// 数值和布尔值
let {toString: m} = 123;
m === Number.prototype.toString; // true
let { toString: n } = true;
n === Boolean.prototype.toString; // true
// 函数参数
function add([o = 1, p]) {
  return o + p;
}
add([0, 2])

# 函数的扩展

// 参数默认值
function add(x = 1, y = 2) {
  console.log(x, y);
}
// 和解构赋值联用
function add({x, y = 2}) {
  console.log(x, y);
} // 通常定义了默认值的参数应该是函数的尾参数，这样比较容易看出来省略了哪些参数，非尾部设置默认值，实际这个参数没法省略
// 作用域：函数进行声明初始化时，参数会形成一个单独的作用域。等到初始化结束，这个作用域就会消失，在不设置参数默认值时不会出现这种语法行为
var x = 1;
function f(x, y = x) {
  console.log(x);
}
f(2); // 2
// rest参数：用于获取函数的多余参数，这样就不需要使用arguments对象了 rest 参数搭配的变量是一个数组，该变量将多余的参数放入数组中。
// rest参数必须是最后一个参数，否则会报错
function add(...values) {
  let sum = 0;
  for (var val of values) {
    sum += val;
  }
  return sum;
}
add(2, 5, 3, 5);
// 箭头函数：函数的简洁表示方法 函数语句多于一句时需要加{} 没有{}，函数语句即为函数返回值 return ****
// 正常函数写法
[1,2,3].map(function (x) {
  return x * x;
});
// 箭头函数写法
[1,2,3].map(x => x * x);
// 注意
// 1.函数体内的this对象，就是定义时所在的对象，而不是使用时所在的对象。
		// 定义对象的方法时如果方法内部使用了this 不建议使用箭头函数定义
// 2.不可以当作构造函数，也就是说，不可以使用new命令，否则会抛出一个错误。
// 3.不可以使用arguments对象，该对象在函数体内不存在。如果要用，可以用 rest 参数代替。
// 4.不可以使用yield命令，因此箭头函数不能用作 Generator 函数。

# 对象的扩展

// 属性的简洁表示法
const prop = 'prop';
const obj = { prop };
const o = {
  method() {
    return 'hello';
  }
}
console.log(obj); // { prop: 'prop' }
// 属性名表达式
let propKey = 'foo';
let obj = {
  [propKey]: true,
  ['a' + 'bc']: 123
};
console.log(obj);
// 方法的name属性
const person = {
  sayName() {
    console.log('hello!');
  },
};
person.sayName.name   // "sayName"
// super关键字：指向当前对象的原型对象
const proto = {
  foo: 'hello'
};
const obj = {
  foo: 'world',
  find() {
    return super.foo;
  }
};
Object.setPrototypeOf(obj, proto);
obj.find() // "hello"
// 扩展运算符
let { x, y, ...z } = { x: 1, y: 2, a: 3, b: 4 };
z // { a: 3, b: 4 }
let foo = { ...['a', 'b', 'c'] };
foo // {0: "a", 1: "b", 2: "c"}
// 等同于 {...Object(1)}
{...1} // {}
// 等同于 {...Object(true)}
{...true} // {}
{...'hello'} // {0: "h", 1: "e", 2: "l", 3: "l", 4: "o"}

// 链判断运算符 ?.
// 在链式调用的时候判断，左侧的对象是否为null或undefined。如果是的，就不再往下运算，而是返回undefined
a?.[++x]
// 等同于
a == null ? undefined : a[++x]

delete a?.b
// 等同于
a == null ? undefined : delete a.b

(a?.b).c
// 等价于
(a == null ? undefined : a.b).c

// null判断运算符 ??
const showSplashScreen = response.settings.showSplashScreen ?? true; // 左侧为null 或者undefined时才会生效
function Component(props) {
  const enable = props.enabled ?? true;
  // …
}
// 跟链判断运算符?.配合使用
const animationDuration = response.settings?.animationDuration ?? 300;
// 如果response.settings是null或undefined，或者response.settings.animationDuration是null或undefined，就会返回默认值300
// 多个逻辑运算符（??、&&、||）一起使用，必须用括号表明优先级，否则会报错

# Symbol

// 表示独一无二的值，可以用作对象属性的名字
let s = Symbol();
typeof s
// "symbol"
// 添加参数对Symbol加以描述以便区分
let s1 = Symbol('foo');
let s2 = Symbol('bar');
s1 // Symbol(foo)
s2 // Symbol(bar)
s1.toString() // "Symbol(foo)"
s2.toString() // "Symbol(bar)"
// 参数是对象会调用的对象的toString方法
const obj = {
  toString() {
    return 'abc';
  }
};
const sym = Symbol(obj);
sym // Symbol(abc)
// Symbol 值不能与其他类型的值进行运算，会报错

// Symbol 作为属性名，遍历对象的时候，该属性不会出现在for...in、for...of循环中，也不会被Object.keys()、Object.getOwnPropertyNames()、JSON.stringify()返回
// 有一个Object.getOwnPropertySymbols()方法，可以获取指定对象的所有 Symbol 属性名

# Set 和 Map

// Set: 类似于数组，但是成员的值都是唯一的，没有重复的值
const s = new Set();
[2, 3, 5, 4, 5, 2, 2].forEach(x => s.add(x));
for (let i of s) {
  console.log(i);
} // 2 3 5 4
// WeakSet: 与Set结构类似，也是不重复的值的集合；区别是：WeakSet的成员只能是对象且不可遍历
// Map: ES6 提供了 Map 数据结构。它类似于对象，也是键值对的集合，但是“键”的范围不限于字符串，各种类型的值（包括对象）都可以当作键。也就是说，Object 结构提供了“字符串—值”的对应，Map 结构提供了“值—值”的对应，是一种更完善的 Hash 结构实现。如果你需要“键值对”的数据结构，Map 比 Object 更合适。
const m = new Map();
const o = {p: 'Hello World'};
m.set(o, 'content')
m.get(o) // "content"
m.has(o) // true
m.delete(o) // true
m.has(o) // false
// WeakMap: 与Map结构类似，也是用于生成键值对的集合 区别是：只接受对象作为键名（null除外）、键名所指向的对象，不计入垃圾回收机制

# async函数

// async函数返回一个 Promise 对象，可以使用then方法添加回调函数。当函数执行的时候，一旦遇到await就会先返回，等到异步操作完成，再接着执行函数体内后面的语句。
function timeout(ms) {
  return new Promise((resolve) => {
    setTimeout(resolve, ms);
  });
}
async function asyncPrint(value, ms) {
  await timeout(ms);
  console.log(value);
}
asyncPrint('hello world', 50);
// async函数内部return语句返回的值，会成为then方法回调函数的参数
// 正常情况下，await命令后面是一个 Promise 对象，返回该对象的结果。如果不是 Promise 对象，就直接返回对应的值。定义了then方法的函数，await会将其视为Promise处理
// 如果await后面的异步操作出错，那么等同于async函数返回的 Promise 对象被reject
async function f() {
  await new Promise(function (resolve, reject) {
    throw new Error('出错了');
  });
}
f()
.then(v => console.log(v))
.catch(e => console.log(e))
// Error：出错了
