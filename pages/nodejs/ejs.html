<!-- 标签含义解析 -->

<!--
  <% '脚本' 标签，用于流程控制，无输出。
  <%_ 删除其前面的空格符
  <%= 输出数据到模板（输出是转义 HTML 标签）
  <%- 输出非转义的数据到模板
  <%# 注释标签，不执行、不输出内容
  <%% 输出字符串 '<%'
  %> 一般结束标签
  -%> 删除紧随其后的换行符
  _%> 将结束标签后面的空格符删除

  注意：配置参数内的delimiter可以修改标签的开闭符号，默认为%。
-->
<template>
  <!-- include指令将相对于模板路径中的模板片段包含进来 -->
  <%- include('./layouts/header') %> <% if (user) { %>
  <h2><%= user.name %></h2>
  <% } %>
</template>

<script setup>
  import { ref } from "vue";
  const user = ref(<%- JSON.stringify(user) %>);
</script>

<style lang="scss" scoped>
  h2 {
    color: red;
  }
</style>

javascript
const path = require("path");
const fs = require("fs");
const ejs = require("ejs");
// 配置参数及说明
const options = {
  cache: true, // 缓存编译后的函数，需要指定 filename
  filename: "", //
  client: true, // 返回独立的编译后的函数
  context: {}, // 函数执行时的上下文
  compileDebug: false, // 当值为false时不编译调试语句
  delimiter: "%", // 放在角括号中的字符，用于标记标签的开与闭
  debug: false, // 将生成的函数体输出
  _with: true, // 是否使用with(){}语句。如果值为false，所有局部数据将存储在 locals 对象中
  localsName: "locals", // 如果不使用 with ，localsName 将作为存储局部变量的对象的名称。默认名称是 locals
  rmWhitespace: false, // 删除所有可安全删除的空白字符，包括开始与结尾处的空格。对于所有标签来说，它提供了一个更安全版本的 -%> 标签（在一行的中间并不会剔除标签后面的换行符)。
  escape: true, // 为 <%= 结构设置对应的转义（escape）函数。它被用于输出结果以及在生成的客户端函数中通过 .toString() 输出。(默认转义 XML)。
  outputFunctionName: "echo", // 设置为代表函数名的字符串（例如 'echo' 或 'print'）时，将输出脚本标签之间应该输出的内容。
  async: false, // 当值为 true 时，EJS 将使用异步函数进行渲染。（依赖于 JS 运行环境对 async/await 是否支持）
};

const tplPath = path.resolve(__dirname, "./template.ejs");

const tpl = fs.readFileSync(path.resolve(__dirname, "./template.ejs"), "utf-8");

const html = ejs.render(tpl, { user: { name: "user1" } }, options); // => 输出渲染后的 HTML 字符串
console.log(html);

const template = ejs.compile(tpl, options); // => 返回一个渲染函数
const html2 = template({ user: { name: "user2" } }); // => 输出渲染后的 HTML 字符串
console.log(html2);

// 没有文件系统的访问权限时，ejs.renderFile无法正常工作
ejs.renderFile(tplPath, { user: { name: "user3" }, options }, (err, html3) => {
  // => 异步渲染
  // html3 => 输出渲染后的 HTML 字符串
  console.log(html3);
});

fs.writeFileSync(path.resolve(__dirname, "./template.vue"), html);
