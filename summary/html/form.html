            # input

            type === 'text'

            <input type="text" placeholder="请输入文字">

            type === 'color'

            <input type="color" placeholder="请选择颜色">

            type === 'password'

            <input type="password" placeholder="请输入密码">

            type === 'checkbox'

            <input type="checkbox">

            type === 'radio'

            <input type="radio">

            type === 'number'

            <input type="number">

            type === 'range'

            <input type="range">

            type === search

            <input type="search">

            type === 'date' || type === 'datetime' || type === 'datetime-local'

            <input type="date">
<input type="datetime">
<input type="datetime-local">

            type === 'month' || type === 'week' || type === 'time'

            <input type="month">
<input type="time">
<input type="week">

            type === 'tel' || type === 'email' || type === 'url'

            <input type="email">
<input type="tel">
<input type="url">

            type === 'button' || type === 'submit'

            <input type="submit" value="submit">
<input type="button" value="click Me!">
